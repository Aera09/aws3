# Program to display numbers from 1 to 100
# Skip numbers divisible by 5
# Stop if number > 80
# Use pass where needed

for num in range(1, 101):

    if num == 1:
        pass    # pass used here as a placeholder, no action needed

    if num % 5 == 0:   # Skip numbers divisible by 5
        continue

    if num > 80:       # Stop the loop
        break

    print(num)

# Program using for loop to print multiplication table.
n = 5
for i in range(1, 11):
    print(n, "x", i, "=", n * i)

# Program using while loop to print the sum of first N numbers.
n = 10
i = 1
s = 0
while i <= n:
    s += i
    i += 1
print("Sum =", s)

# Program to print a pattern using nested loops.
*
**
***
****
for i in range(1, 5):
    for j in range(i):
        print("*", end="")
    print()

# Program to check prime number using loop.
n = 17
flag = True
for i in range(2, n):
    if n % i == 0:
        flag = False
        break
if flag:
    print("Prime")
else:
    print("Not Prime")

# Program to reverse a number using while loop.
n = 1234
rev = 0
while n > 0:
    digit = n % 10
    rev = rev * 10 + digit
    n //= 10
print("Reversed:", rev)

# Nested loop to print numbers in matrix form.
1 2 3
4 5 6
7 8 9
num = 1
for i in range(3):
    for j in range(3):
        print(num, end=" ")
        num += 1
    print()
