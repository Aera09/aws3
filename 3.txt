# Program to display numbers from 1 to 100
# Skip numbers divisible by 5
# Stop if number > 80
# Use pass where needed

for num in range(1, 101):

    if num == 1:
        pass    # pass used here as a placeholder, no action needed

    if num % 5 == 0:   # Skip numbers divisible by 5
        continue

    if num > 80:       # Stop the loop
        break

    print(num)
